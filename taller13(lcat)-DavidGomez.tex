\sect{Punto 2}
\begin{logicenv}{Todos los latinos\dots}
    \begin{gather*}
        L(x) := \text{``$x$ es latino''}\\
        M(x) := \text{``$x$ es mediterraneo''}\\
        N(x) := \text{``$x$ es nórdico''}
    \end{gather*}
    \begin{logic}
        (\forall x \,\vert\, L(x) : M(x)) & Suposición\\
        (\forall x \,\vert\, N(x) : \neg M(x)) & Suposición\\
        (\forall x \,\vert:\, (\neg L(x) \lor M(x)) \land (\neg N(x) \lor \neg M(x))) & Unión, trueque, dist.($\forall x$, $\land$)\\
        (\forall x \,\vert:\, \neg L(x) \lor \neg N(x)) & Corte\\
        (\forall x \,\vert\, N(x) : \neg L(x)) & trueque
    \end{logic}
\end{logicenv}
